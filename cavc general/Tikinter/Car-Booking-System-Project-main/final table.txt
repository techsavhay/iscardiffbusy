CREATE TABLE Admin
(
 A_id number primary key,
 Username varchar2(10) not null unique,
 Password varchar2(30) not null);


CREATE TABLE Car_desc
(
 Car_id number primary key,
 plateNo VARCHAR2(10) not null unique,
 Model VARCHAR2(15),
 Capacity NUMBER,
 Milage number,
 Name varchar2(15),
 Review number
);


CREATE TABLE Customer
(
 C_id number primary key,
 contact number,
 Address varchar2(25),
 Username varchar2(10) not null unique,
 Password varchar2(30) not null
);


CREATE TABLE Rental
(
 R_id number primary key,
 Rental_Date DATE,
 Return_Date DATE,
 Car_id number,
 C_id number,
 Foreign key (Car_id) references Car_desc(Car_id) ON DELETE SET NULL,
 Foreign key (C_id) references Customer(C_id)
 ON DELETE SET NULL
);


CREATE TABLE Payment
(
 P_id number primary key,
 A_id number,
 C_id number,
 R_id number,
 Pay_date date,
 amount number,
 Foreign key (A_id) references Admin(A_id) ON DELETE SET NULL,
 Foreign key (R_id) references Rental(R_id),
 Foreign key (C_id) references Customer(C_id)
 ON DELETE SET NULL
);

CREATE OR REPLACE Trigger Pass_A
BEFORE INSERT OR UPDATE ON Admin
FOR EACH ROW
BEGIN
 IF (length(:new.Password)<8) THEN
  RAISE_APPLICATION_ERROR(-20500,'Password must be atleast 8 characters');
 END IF;
END;
/


CREATE OR REPLACE Trigger Pass_C
BEFORE INSERT OR UPDATE ON Customer
FOR EACH ROW
BEGIN
 IF (length(:new.Password)<8) THEN
  RAISE_APPLICATION_ERROR(-20500,'Password must be atleast 8 characters');
 END IF;
END;
/

START WITH 1
